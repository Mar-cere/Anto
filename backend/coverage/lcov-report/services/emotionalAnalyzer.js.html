    this.diminishersPattern = /(?:poco|algo|ligeramente|un poco|apenas|levemente|ligeramente|medianamente|moderadamente|relativamente|bastante.*poco|no.*mucho|no.*tan|no.*muy|no.*demasiado|un.*poco.*solo|solo.*un.*poco)/i;
    this.contextualCluesPattern = /(?:me siento|estoy|siento que|creo que|pienso que|me parece que|siento como|me.*siento|estoy.*sintiendo|me.*estoy.*sintiendo|siento.*como.*si|me.*parece|me.*da.*la.*sensacin|tengo.*la.*sensacin|me.*hace.*sentir|me.*hace.*estar|me.*provoca|me.*genera|me.*causa|me.*produce)/i;
    
    // Patrones de negacin emocional (indican emocin negativa aunque no se mencione directamente)
    this.negativeIndicatorsPattern = /(?:no.*me.*siento.*bien|no.*estoy.*bien|no.*me.*va.*bien|no.*puedo|no.*puedo.*ms|no.*aguanto|no.*soporto|no.*tengo.*ganas|no.*me.*motiva|no.*me.*gusta|no.*me.*agrada|no.*me.*sirve|no.*funciona|no.*sale.*bien|nada.*funciona|todo.*sale.*mal|nada.*me.*sale.*bien|todo.*me.*sale.*mal|nunca.*funciona|siempre.*sale.*mal)/i;
    
    // Patrones de negacin emocional explcita (cuando se niega una emocin, puede indicar que s la siente)
    this.denialPattern = /(?:no.*estoy.*triste|no.*me.*siento.*triste|no.*estoy.*deprimid(?:o|a)|no.*estoy.*ansios(?:o|a)|no.*me.*siento.*ansios(?:o|a)|no.*estoy.*enojad(?:o|a)|no.*me.*siento.*enojad(?:o|a)|no.*tengo.*miedo|no.*me.*da.*miedo|no.*estoy.*preocupad(?:o|a)|no.*me.*preocupa)/i;
    
    // Patrones de comparacin temporal (mejor/peor que antes)
    this.temporalComparisonPattern = /(?:mejor.*que|peor.*que|igual.*que|ms.*que|menos.*que|antes.*estaba|ahora.*estoy|antes.*me.*senta|ahora.*me.*siento|me.*siento.*mejor|me.*siento.*peor|estoy.*mejor|estoy.*peor)/i;
    
    // Patrones de preguntas retricas (indican frustracin o preocupacin)
    this.rhetoricalQuestionPattern = /(?:por.*qu.*siempre|por.*qu.*nunca|por.*qu.*a.*m|qu.*hice.*yo|qu.*hice.*mal|por.*qu.*me.*pasa|por.*qu.*a.*m.*siempre|qu.*tengo.*que.*hacer|qu.*puedo.*hacer|qu.*debo.*hacer|por.*qu.*todo|por.*qu.*nada)/i;
    
    // Emojis emocionales comunes
    this.emojiPatterns = {
      tristeza: /[]/,
      ansiedad: /[]/,
      enojo: /[]/,
      alegria: /[]/,
      miedo: /[]/,
      verguenza: /[]/,
      culpa: /[]/,
      esperanza: /[]/,
      neutral: /[]/
    };
    
    // Patrones de repeticin (indican mayor intensidad)
    this.repetitionPattern = /(?:muy.*muy|mucho.*mucho|demasiado.*demasiado|super.*super|extremadamente.*extremadamente)/i;
    
    // Patrones de signos de puntuacin mltiples (indican intensidad)
    this.exclamationPattern = /!{2,}/;
    this.questionPattern = /\?{2,}/;
    
    // Patrones de maysculas (indican intensidad o urgencia)
    this.allCapsPattern = /^[A-Z\s]{10,}$/;
  }
  
  // Helper: validar que el contenido es un string vlido
  isValidString(value) {
    return typeof value === 'string' && value.trim().length > 0;
  }
  
  // Helper: obtener emocin neutral por defecto
  getNeutralEmotion() {
    return {
      name: this.EMOTION_NEUTRAL,
      category: this.CATEGORY_NEUTRAL,
      baseIntensity: this.INTENSITY_DEFAULT
    };
  }
  
  // Helper: limitar intensidad dentro del rango vlido
  clampIntensity(intensity) {
    return Math.max(this.INTENSITY_MIN, Math.min(intensity, this.INTENSITY_MAX));
  }
 
  /**
   * Analiza la emocin principal y secundaria de un mensaje
   * @param {string} content - Contenido del mensaje
   * @param {Array} previousPatterns - Historial de anlisis previos (opcional)
   * @returns {Promise<Object>} Anlisis emocional con emocin principal, intensidad, categora, emociones secundarias, confianza y si requiere atencin
   */
  async analyzeEmotion(content, previousPatterns = []) {
    if (!this.isValidString(content)) {
      console.warn('[EmotionalAnalyzer] Contenido invlido recibido en analyzeEmotion:', content);
      return this.getDefaultAnalysis();
    }
 
    try {
      // NUEVO: Verificar cach antes de analizar
      let cachedAnalysis = null;
      try {
        const { default: analysisCache } = await import('./emotionalAnalysisCache.js');
        cachedAnalysis = analysisCache.get(content);
        
        if (cachedAnalysis && (!previousPatterns || previousPatterns.length === 0)) {
          // Si hay anlisis cacheado y no hay patrones previos, usar cach
          return cachedAnalysis;
        }
      } catch (error) {
        // Si falla el cach, continuar con anlisis normal
        console.warn('[EmotionalAnalyzer] Error al acceder al cach:', error);
      }
 
      const contentLower = content.toLowerCase();
      let detectedEmotion = this.detectPrimaryEmotion(contentLower);
      let intensity = this.calculateIntensity(contentLower, detectedEmotion);
      let secondaryEmotions = this.detectSecondaryEmotions(contentLower, detectedEmotion.name);
 
      // Ajustar basado en patrones previos si existen
      if (previousPatterns && Array.isArray(previousPatterns) && previousPatterns.length > 0) {
        const adjustedAnalysis = this.adjustBasedOnHistory(
          detectedEmotion,
          intensity,
          previousPatterns
        );
        detectedEmotion = adjustedAnalysis.emotion;
        intensity = adjustedAnalysis.intensity;
      }
 
      // NUEVO: Detectar subtipo emocional
      let subtype = null;
      try {
        const { default: subtypeDetector } = await import('./emotionalSubtypeDetector.js');
        subtype = subtypeDetector.detectSubtype(detectedEmotion.name, content);
      } catch (error) {
        console.warn('[EmotionalAnalyzer] Error al detectar subtipo:', error);
      }
 
      // NUEVO: Detectar tema/contexto del mensaje
      let topic = 'general';
      let topics = ['general'];
      try {
        const { default: topicDetector } = await import('./topicDetector.js');
        topic = topicDetector.detectTopic(content);
        topics = topicDetector.detectMultipleTopics(content);
      } catch (error) {
        console.warn('[EmotionalAnalyzer] Error al detectar tema:', error);
      }
 
      const analysis = {
        mainEmotion: detectedEmotion.name,
        intensity,
        category: detectedEmotion.category,
        secondary: secondaryEmotions,
        confidence: this.calculateConfidence(contentLower, detectedEmotion),
        requiresAttention: this.checkIfRequiresAttention(detectedEmotion, intensity),
        // NUEVOS CAMPOS
        subtype: subtype, // Subtipo emocional (micro-emocin)
        topic: topic, // Tema principal del mensaje
        topics: topics // Mltiples temas detectados
      };
 
      // NUEVO: Guardar en cach si no hay patrones previos
      if (!previousPatterns || previousPatterns.length === 0) {
        try {
          const { default: analysisCache } = await import('./emotionalAnalysisCache.js');
          analysisCache.set(content, analysis);
        } catch (error) {
          // Si falla, continuar sin cach
        }
      }
 
      return analysis;
    } catch (error) {
      console.error('[EmotionalAnalyzer] Error en anlisis emocional:', error, content);
      return this.getDefaultAnalysis();
    }
  }
 
  /**
   * Detecta emojis en el contenido y retorna la emocin asociada
   * @param {string} content - Contenido del mensaje
   * @returns {string|null} Nombre de la emocin detectada por emoji o null
   */
  detectEmotionFromEmojis(content) {
    if (!this.isValidString(content)) {
      return null;
    }
    
    // Buscar emojis en orden de prioridad (emociones ms intensas primero)
    // IMPORTANTE: Priorizar emojis positivos si hay texto positivo, y negativos si hay texto negativo
    const hasPositiveText = /(?:genial|feliz|contento|alegre|bien|bueno)/i.test(content);
    const hasNegativeText = /(?:mal|triste|malo|deprimido|ansioso)/i.test(content);
    
    // Si hay texto positivo, priorizar emojis positivos
    if (hasPositiveText && !hasNegativeText) {
      const positiveEmotions = ['alegria', 'esperanza', 'neutral'];
      for (const emotion of positiveEmotions) {
        if (this.emojiPatterns[emotion] && this.emojiPatterns[emotion].test(content)) {
          return emotion;
        }
      }
    }
    
    // Si hay texto negativo, priorizar emojis negativos
    if (hasNegativeText && !hasPositiveText) {
      const negativeEmotions = ['tristeza', 'ansiedad', 'miedo', 'enojo', 'verguenza', 'culpa'];
      for (const emotion of negativeEmotions) {
        if (this.emojiPatterns[emotion] && this.emojiPatterns[emotion].test(content)) {
          return emotion;
        }
      }
    }
    
    // Orden por defecto (emociones ms intensas primero)
    const emotionPriority = ['enojo', 'miedo', 'tristeza', 'ansiedad', 'verguenza', 'culpa', 'alegria', 'esperanza', 'neutral'];
    
    for (const emotion of emotionPriority) {
      if (this.emojiPatterns[emotion] && this.emojiPatterns[emotion].test(content)) {
        return emotion;
      }
    }
    
    return null;
  }
 
  /**
   * Detecta la emocin principal en el contenido
   * @param {string} content - Contenido en minsculas
   * @returns {Object} Emocin detectada con nombre, categora e intensidad base
   */
  detectPrimaryEmotion(content) {
    if (!this.isValidString(content)) {
      return this.getNeutralEmotion();
    }
    
    // Primero verificar emojis (tienen alta confianza)
    const emojiEmotion = this.detectEmotionFromEmojis(content);
    if (emojiEmotion && emojiEmotion !== 'neutral') {
      const emotionData = this.emotionPatterns[emojiEmotion];
      return {
        name: emojiEmotion,
        category: emotionData.category,
        baseIntensity: emotionData.intensity
      };
    }
    
    // Verificar negacin emocional explcita (puede indicar que s siente esa emocin)
    if (this.denialPattern.test(content)) {
      // Extraer la emocin negada
      if (/no.*triste|no.*deprimid/i.test(content)) {
        return {
          name: 'tristeza',
          category: 'negative',
          baseIntensity: 5 // Intensidad menor porque est negada
        };
      } else if (/no.*ansios|no.*preocupa/i.test(content)) {
        return {
          name: 'ansiedad',
          category: 'negative',
          baseIntensity: 5
        };
      } else if (/no.*enojad/i.test(content)) {
        return {
          name: 'enojo',
          category: 'negative',
          baseIntensity: 5
        };
      } else if (/no.*miedo/i.test(content)) {
        return {
          name: 'miedo',
          category: 'negative',
          baseIntensity: 5
        };
      }
    }
    
    // Verificar preguntas retricas (indican frustracin o ansiedad)
    if (this.rhetoricalQuestionPattern.test(content)) {
      const hasAnxietyKeywords = /(?:preocupa|nervios|inquiet|agobia|abruma|tensin)/i.test(content);
      if (hasAnxietyKeywords) {
        return {
          name: 'ansiedad',
          category: 'negative',
          baseIntensity: 6
        };
      } else {
        return {
          name: 'enojo',
          category: 'negative',
          baseIntensity: 7
        };
      }
    }
    
    // Buscar emociones especficas por patrones de texto
    // IMPORTANTE: Verificar primero si hay negaciones explcitas que invalidan emociones positivas
    // Por ejemplo: "no me gusta" no debe detectarse como "alegria"
    const hasNegativePrefix = /^(?:no|nunca|jams|tampoco)\s+/i.test(content.trim());
    
    // Priorizar emociones positivas primero, PERO solo si no hay prefijo negativo
    // IMPORTANTE: Priorizar miedo antes que ansiedad para evitar confusin
    // IMPORTANTE: Priorizar neutral para mensajes simples como "estoy normal"
    // IMPORTANTE: NO considerar mensajes simples si tienen comparaciones temporales negativas
    // IMPORTANTE: Si hay contexto de autolesin o crisis, priorizar emociones negativas
    const hasTemporalComparison = this.temporalComparisonPattern.test(content);
    const hasNegativeComparison = /(?:peor|ms.*mal|menos.*bien)/i.test(content);
    const hasSelfHarmContext = /(?:dolor.*fsico|dolor.*emocional|hacerme.*dao|autolesi|me.*corto|me.*quemo|me.*golpeo|hacerse.*dao|sera.*mejor.*si.*no.*existiera|mejor.*si.*no.*existiera|pienso.*que.*sera.*mejor.*si.*no.*existiera|terminar.*con.*todo.*esto|pensado.*en.*cmo.*terminar|nadie.*me.*extraara.*si.*me.*fuera|me.*extraara.*si.*me.*fuera|no.*veo.*ninguna.*salida|sin.*salida|futuro.*se.*ve.*completamente.*negro|futuro.*negro|ya.*no.*puedo.*ms.*estoy.*al.*lmite|estoy.*al.*lmite|estoy.*hecho.*polvo|hecho.*polvo|completamente.*destrozado|destrozado)/i.test(content);
    const hasCrisisContext = /(?:ataque.*de.*pnico|ataque.*de.*ansiedad|no.*puedo.*respirar|me.*ahogo|palpitaciones|siento.*que.*me.*voy.*a.*morir|siento.*que.*no.*estoy.*en.*mi.*cuerpo|como.*si.*fuera.*un.*sueo|no.*estoy.*en.*mi.*cuerpo|no.*puedo.*calmarme)/i.test(content);
    const isSimpleMessage = /^(estoy|me siento|soy|est|estn)\s+\w+$/i.test(content.trim()) && !hasNegativeComparison;
    
    // Si hay contexto de autolesin o crisis, priorizar emociones negativas
    const emotionPriority = (hasSelfHarmContext || hasCrisisContext)
      ? ['tristeza', 'ansiedad', 'miedo', 'enojo', 'verguenza', 'culpa', 'alegria', 'esperanza', 'neutral']
      : isSimpleMessage 
        ? ['neutral', 'alegria', 'esperanza', 'miedo', 'tristeza', 'ansiedad', 'enojo', 'verguenza', 'culpa']
        : ['alegria', 'esperanza', 'miedo', 'tristeza', 'ansiedad', 'enojo', 'verguenza', 'culpa', 'neutral'];
    
    // Patrones para casos especiales
    const sarcasmPositivePattern = /(?:?qu genial!?[^a-zA-Z0-9]*otro problema ms|otro problema ms)/i;
    const mixedTemporalPattern = /me.*siento.*mejor.*que.*hace.*una.*semana.*pero.*peor.*que.*ayer/i;
    const cycleTemporalPattern = /algunos.*d[i]as.*estoy.*bien.*otros.*d[i]as.*muy.*mal/i;
    const emojisNegative = /[\u{1F622}\u{1F62D}\u{1F61E}\u{1F625}\u{1F614}\u{2639}\u{1F641}]/u;
    const textoMuyPositivo = /(?:todo.*est.*perfecto|todo.*perfecto|est.*perfecto)/i;
 
    // Caso especial antes de todo: texto muy positivo + emoji muy negativo => tristeza
    if (emojisNegative.test(content) && textoMuyPositivo.test(content)) {
      return {
        name: 'tristeza',
        category: 'negative',
        baseIntensity: 7
      };
    }
 
    // Primero buscar emociones prioritarias
    for (const emotion of emotionPriority) {
      if (this.emotionPatterns[emotion] && this.emotionPatterns[emotion].patterns.test(content)) {
        const emotionData = this.emotionPatterns[emotion];
        
        // Si es una emocin positiva y hay un prefijo negativo, no aplicarla
        // Ejemplo: "no me gusta" no debe ser "alegria"
        if (emotionData.category === 'positive' && hasNegativePrefix) {
          // Verificar si el patrn especfico est precedido por "no"
          const positivePatterns = ['me.*gusta', 'me.*encanta', 'me.*alegra', 'me.*emociona'];
          const hasPositivePattern = positivePatterns.some(pattern => {
            const regex = new RegExp(pattern, 'i');
            return regex.test(content);
          });
          
          if (hasPositivePattern && /no\s+me/i.test(content)) {
            // Saltar esta emocin positiva, continuar buscando
            continue;
          }
        }
        
        // IMPORTANTE: Si hay contexto de autolesin o crisis, priorizar emociones negativas
        // Ejemplo: "El dolor fsico me hace sentir mejor que el dolor emocional" debe ser tristeza, no alegria
        const hasSelfHarmContextForPositive = /(?:dolor.*fsico|dolor.*emocional|hacerme.*dao|autolesi|me.*corto|me.*quemo|me.*golpeo|hacerse.*dao)/i.test(content);
        if (emotionData.category === 'positive' && hasSelfHarmContextForPositive) {
          // Si hay contexto de autolesin, no aplicar emocin positiva
          continue;
        }
 
        // Caso especial: sarcasmo con frase aparentemente positiva
        if (emotionData.category === 'positive' && sarcasmPositivePattern.test(content)) {
          return {
            name: 'enojo',
            category: 'negative',
            baseIntensity: 7
          };
        }
 
        // Caso especial: ciclo temporal "algunos das bien, otros muy mal"
        if (cycleTemporalPattern.test(content) && emotion === 'alegria') {
          return {
            name: 'tristeza',
            category: 'negative',
            baseIntensity: 7
          };
        }
 
        // Caso especial: texto muy positivo con emoji claramente negativo -> tristeza
        if (emotion === 'alegria' && textoMuyPositivo.test(content) && emojisNegative.test(content)) {
          return {
            name: 'tristeza',
            category: 'negative',
            baseIntensity: 7
          };
        }
        
        return {
          name: emotion,
          category: emotionData.category,
          baseIntensity: emotionData.intensity
        };
      }
    }
    
    // Luego buscar otras emociones
    for (const [emotion, data] of Object.entries(this.emotionPatterns)) {
      if (emotion !== 'neutral' && !emotionPriority.includes(emotion) && data.patterns.test(content)) {
        // Misma validacin para prefijos negativos
        if (data.category === 'positive' && hasNegativePrefix) {
          const positivePatterns = ['me.*gusta', 'me.*encanta', 'me.*alegra', 'me.*emociona'];
          const hasPositivePattern = positivePatterns.some(pattern => {
            const regex = new RegExp(pattern, 'i');
            return regex.test(content);
          });
          
          if (hasPositivePattern && /no\s+me/i.test(content)) {
            continue;
          }
        }
        
        // IMPORTANTE: Si hay contexto de autolesin o crisis, priorizar emociones negativas
        const hasSelfHarmContext = /(?:dolor.*fsico|dolor.*emocional|hacerme.*dao|autolesi|me.*corto|me.*quemo|me.*golpeo|hacerse.*dao)/i.test(content);
        if (data.category === 'positive' && hasSelfHarmContext) {
          // Si hay contexto de autolesin, no aplicar emocin positiva
          continue;
        }
        
        return {
          name: emotion,
          category: data.category,
          baseIntensity: data.intensity
        };
      }
    }
    
    // Si no se detecta una emocin especfica, verificar indicadores negativos
    // Estos pueden indicar tristeza o ansiedad aunque no se mencione explcitamente
    if (this.negativeIndicatorsPattern.test(content)) {
      // Verificar primero casos crticos de ideacin suicida y desesperanza
      const hasSuicidalIdeation = /(?:sera.*mejor.*si.*no.*existiera|mejor.*si.*no.*existiera|pienso.*que.*sera.*mejor.*si.*no.*existiera|terminar.*con.*todo.*esto|pensado.*en.*cmo.*terminar|nadie.*me.*extraara.*si.*me.*fuera|me.*extraara.*si.*me.*fuera)/i.test(content);
      const hasHopelessness = /(?:no.*veo.*ninguna.*salida|sin.*salida|futuro.*se.*ve.*completamente.*negro|futuro.*negro)/i.test(content);
      const hasDesperation = /(?:ya.*no.*puedo.*ms.*estoy.*al.*lmite|estoy.*al.*lmite)/i.test(content);
      const hasDisconnection = /(?:me.*siento.*desconectado.*de.*todos|desconectado.*de.*todos)/i.test(content);
      const hasColloquialNegative = /(?:estoy.*hecho.*polvo|hecho.*polvo)/i.test(content);
      const hasExtremeNegative = /(?:completamente.*destrozado|destrozado)/i.test(content);
      
      if (hasSuicidalIdeation || hasHopelessness || hasDesperation) {
        return {
          name: 'tristeza',
          category: 'negative',
          baseIntensity: 9
        };
      }
      
      if (hasDisconnection) {
        return {
          name: 'tristeza',
          category: 'negative',
          baseIntensity: 7
        };
      }
      
      if (hasColloquialNegative || hasExtremeNegative) {
        return {
          name: 'tristeza',
          category: 'negative',
          baseIntensity: 8
        };
      }
      
      // Determinar si es ms tristeza o ansiedad basado en el contexto
      const hasAnxietyKeywords = /(?:preocupa|nervios|inquiet|agobia|abruma|tensin|me.*aterra.*pensar|aterroriza.*pensar)/i.test(content);
      const hasSadnessKeywords = /(?:sin.*ganas|desmotiva|desnimo|vaco|hundid)/i.test(content);
      
      if (hasAnxietyKeywords) {
        return {
          name: 'ansiedad',
          category: 'negative',
          baseIntensity: 6
        };
      } else if (hasSadnessKeywords) {
        return {
          name: 'tristeza',
          category: 'negative',
          baseIntensity: 7
        };
      } else {
        // Por defecto, indicadores negativos sugieren tristeza
        return {
          name: 'tristeza',
          category: 'negative',
          baseIntensity: 6 // Intensidad ligeramente menor si no es explcita
        };
      }
    }
    
    // Si no hay indicadores, retornar neutral
    return this.getNeutralEmotion();
  }
 
  /**
   * Calcula la intensidad emocional ajustada segn modificadores en el contenido
   * @param {string} content - Contenido del mensaje
   * @param {Object} emotion - Emocin detectada
   * @returns {number} Intensidad ajustada (1-10)
   */
  calculateIntensity(content, emotion) {
    if (!this.isValidString(content) || !emotion) {
      return this.INTENSITY_DEFAULT;
    }
    
    let intensity = emotion.baseIntensity;
    
    // ========== AJUSTES ESPECIALES PARA CRISIS Y AUTOLESIN ==========
    
    // Patrones de crisis de pnico (alta intensidad)
    const panicPatterns = /(?:ataque.*de.*pnico|ataque.*de.*ansiedad|no.*puedo.*respirar|me.*ahogo|mi.*corazn.*late.*muy.*rpido|palpitaciones|siento.*que.*me.*voy.*a.*morir|siento.*que.*me.*desmayo|prdida.*de.*control|no.*puedo.*pensar.*claramente|no.*puedo.*calmarme|siento.*que.*no.*soy.*yo|fuera.*de.*mi.*cuerpo|despersonalizacin|siento.*que.*no.*estoy.*en.*mi.*cuerpo|como.*si.*fuera.*un.*sueo|no.*estoy.*en.*mi.*cuerpo)/i;
    if (panicPatterns.test(content) && emotion.name === 'ansiedad') {
      intensity = Math.max(this.clampIntensity(intensity + 3), 9); // Aumentar significativamente, mnimo 9
    }
 
    // Miedo anticipatorio ("me aterra pensar...") debe ser al menos 8
    if (emotion.name === 'miedo' && /me.*aterra.*pensar|aterroriza.*pensar/i.test(content)) {
      intensity = Math.max(intensity, 8);
    }
    
    // Patrones de autolesin (alta intensidad)
    // IMPORTANTE: Verificar primero autolesin pasada para evitar falsos positivos
    const pastSelfHarmPatterns = /(?:cicatrices.*de.*cuando.*me.*cortaba|cuando.*me.*cortaba|cuando.*me.*quemaba|cuando.*me.*golpeaba|de.*cuando.*me.*cortaba|me.*cortaba.*antes|me.*quemaba.*antes)/i;
    const isPastSelfHarm = pastSelfHarmPatterns.test(content);
    
    if (isPastSelfHarm && emotion.name === 'tristeza') {
      // Autolesin pasada: intensidad moderada-alta (7-9), no mxima
      // Establecer intensidad base para autolesin pasada (no aplicar otros ajustes de autolesin)
      intensity = this.clampIntensity(7 + 1); // Base 7 + 1 = 8, mximo 9 con otros ajustes
    } else if (emotion.name === 'tristeza') {
      // Patrones de alta urgencia (intensidad mxima)
      // Incluir tambin expresiones que indican que es la nica forma de sentir algo
      const urgentSelfHarmPatterns = /(?:tengo.*ganas.*de.*hacerme.*dao|quiero.*hacerme.*dao|me.*quiero.*cortar|quiero.*cortarme|necesito.*hacerme.*dao|urgencia.*de.*hacerme.*dao|no.*puedo.*dejar.*de.*pensar.*en.*hacerme.*dao|pensar.*en.*hacerme.*dao|es.*la.*nica.*forma.*de.*sentir|nica.*forma.*de.*sentir.*algo)/i;
      if (urgentSelfHarmPatterns.test(content)) {
        // Para autolesin urgente, establecer intensidad mnima de 9
        intensity = Math.max(this.clampIntensity(intensity + 3), 9);
      }
      
      // Patrones de autolesin activa (alta intensidad)
      const activeSelfHarmPatterns = /(?:me.*corto(?!.*cuando)|me.*quemo(?!.*cuando)|me.*golpeo(?!.*cuando)|autolesi(?:n|on|arme)|el.*dolor.*fsico.*me.*hace.*sentir.*mejor)/i;
      if (activeSelfHarmPatterns.test(content)) {
        if (/a veces/i.test(content)) {
          // Si hay "a veces", intensidad moderada-alta (8-10) pero no mxima
          intensity = Math.max(this.clampIntensity(intensity + 2), 8);
        } else {
          intensity = this.clampIntensity(intensity + 3); // Aumentar significativamente
        }
      }
    }
    
    // Guardar intensidad despus de ajustes especiales para evitar que se reduzca
    const intensityAfterSpecialAdjustments = intensity;
    const hasUrgentSelfHarm = emotion.name === 'tristeza' && /(?:tengo.*ganas.*de.*hacerme.*dao|quiero.*hacerme.*dao|me.*quiero.*cortar|quiero.*cortarme|necesito.*hacerme.*dao|urgencia.*de.*hacerme.*dao|no.*puedo.*dejar.*de.*pensar.*en.*hacerme.*dao|pensar.*en.*hacerme.*dao|es.*la.*nica.*forma.*de.*sentir|nica.*forma.*de.*sentir.*algo)/i.test(content);
    const hasActiveSelfHarmWithAveces = emotion.name === 'tristeza' && /(?:me.*corto|me.*quemo|me.*golpeo)/i.test(content) && /a veces/i.test(content);
    const hasSuicidalIdeation = /(?:sera.*mejor.*si.*no.*existiera|mejor.*si.*no.*existiera|pienso.*que.*sera.*mejor.*si.*no.*existiera|terminar.*con.*todo.*esto|pensado.*en.*cmo.*terminar|nadie.*me.*extraara.*si.*me.*fuera|me.*extraara.*si.*me.*fuera)/i.test(content);
    const hasHopelessness = /(?:no.*veo.*ninguna.*salida|sin.*salida|futuro.*se.*ve.*completamente.*negro|futuro.*negro)/i.test(content);
    const hasDesperation = /(?:ya.*no.*puedo.*ms.*estoy.*al.*lmite|estoy.*al.*lmite)/i.test(content);
    
    // Ajustar segn intensificadores o atenuadores
    // IMPORTANTE: No aplicar atenuadores si hay contexto de autolesin o crisis (son situaciones crticas)
    const hasSelfHarmOrCrisis = /(?:hacerme.*dao|autolesi|ataque.*de.*pnico|no.*puedo.*respirar|me.*ahogo|pensar.*en.*hacerme.*dao|quiero.*cortarme|me.*quemo|me.*corto|me.*golpeo)/i.test(content);
    if (this.intensifiersPattern.test(content)) {
      intensity = this.clampIntensity(intensity + this.INTENSITY_ADJUSTMENT);
    } else if (this.diminishersPattern.test(content) && !hasSelfHarmOrCrisis) {
      // Solo aplicar atenuadores si NO hay contexto de autolesin o crisis
      intensity = this.clampIntensity(intensity - this.INTENSITY_ADJUSTMENT);
    }
    
    // Ajustar segn repeticin de palabras (muy muy, mucho mucho, etc.)
    if (this.repetitionPattern.test(content)) {
      intensity = this.clampIntensity(intensity + 1);
    }
    
    // Ajustar segn signos de exclamacin mltiples
    const exclamationCount = (content.match(/!/g) || []).length;
    if (exclamationCount >= 2) {
      intensity = this.clampIntensity(intensity + Math.min(exclamationCount - 1, 2));
    }
    
    // Ajustar segn signos de interrogacin mltiples (pueden indicar ansiedad o frustracin)
    const questionCount = (content.match(/\?/g) || []).length;
    if (questionCount >= 2 && emotion.category === 'negative') {
      intensity = this.clampIntensity(intensity + 1);
    }
    
    // Ajustar segn maysculas (TODO EN MAYSCULAS indica urgencia/intensidad)
    if (this.allCapsPattern.test(content)) {
      intensity = this.clampIntensity(intensity + 2);
    }
    
    // Ajustar segn longitud del mensaje (mensajes largos pueden indicar mayor intensidad)
    const wordCount = content.split(/\s+/).filter(w => w.length > 0).length;
    if (wordCount > this.WORD_COUNT_THRESHOLD) {
      intensity = this.clampIntensity(intensity + 1);
    }
    
    // Ajustar segn comparaciones temporales
    // IMPORTANTE: No aplicar ajustes de comparacin temporal si hay contexto de autolesin o crisis
    if (this.temporalComparisonPattern.test(content) && !hasSelfHarmOrCrisis) {
      if (/peor.*que|ms.*mal|menos.*bien/i.test(content)) {
        intensity = this.clampIntensity(intensity + 1);
      } else if (/mejor.*que|ms.*bien|menos.*mal/i.test(content) && emotion.category === 'negative') {
        intensity = this.clampIntensity(intensity - 1);
      }
    }
 
    // Ajustes para comparaciones temporales complejas
    const hasMixedTemporal = /me.*siento.*mejor.*que.*hace.*una.*semana.*pero.*peor.*que.*ayer/i.test(content);
    const hasCycleTemporal = /algunos.*d[i]as.*estoy.*bien.*otros.*d[i]as.*muy.*mal/i.test(content);
    if (hasMixedTemporal) {
      intensity = Math.min(intensity, 7);
    }
    if (hasCycleTemporal) {
      intensity = Math.min(intensity, 8);
    }
    
    // IMPORTANTE: Asegurar intensidades mnimas despus de todos los ajustes
    if (hasUrgentSelfHarm || hasSuicidalIdeation) {
      intensity = Math.max(intensity, 9); // Mnimo 9 para autolesin urgente o ideacin suicida
    } else if (hasActiveSelfHarmWithAveces) {
      intensity = Math.max(intensity, 8); // Mnimo 8 para autolesin activa con "a veces"
    }
    
    // Ajustes adicionales para casos especficos
    if (hasHopelessness && emotion.name === 'tristeza') {
      intensity = Math.max(intensity, 8); // Mnimo 8 para desesperanza
    }
    
    if (hasDesperation && emotion.name === 'tristeza') {
      intensity = Math.max(intensity, 9); // Mnimo 9 para desesperacin
    }
    
    // Caso especial: planificacin explcita "terminar con todo esto" debe ser 10
    if (hasSuicidalIdeation && /terminar.*con.*todo.*esto/i.test(content)) {
      intensity = 10;
    }
 
    // Ajuste para culpa y vergenza con intensificadores
    if ((emotion.name === 'culpa' || emotion.name === 'verguenza') &&
        /(?:todo.*es.*mi.*culpa|siempre.*arruino.*todo|mucha.*vergenza|mucha.*verguenza)/i.test(content)) {
      intensity = Math.max(intensity, 8); // Mnimo 8 para culpa/vergenza intensa
    }
 
    // Ajuste para minimizacin suave de tristeza ("solo estoy un poco triste")
    if (emotion.name === 'tristeza' &&
        /(?:solo.*estoy.*un.*poco.*triste|un.*poco.*triste)/i.test(content)) {
      intensity = Math.max(intensity, 5);
    }
    
    return intensity;
  }
 
  /**
   * Detecta emociones secundarias en el contenido (diferentes a la principal)
   * @param {string} content - Contenido del mensaje
   * @param {string} primaryEmotion - Emocin principal detectada
   * @returns {Array} Emociones secundarias detectadas
   */
  detectSecondaryEmotions(content, primaryEmotion) {
    const secondaryEmotions = [];
    
    if (!this.isValidString(content)) {
      return secondaryEmotions;
    }
    
    for (const [emotion, data] of Object.entries(this.emotionPatterns)) {
      if (emotion !== primaryEmotion && data.patterns.test(content)) {
        secondaryEmotions.push(emotion);
      }
    }
    
    return secondaryEmotions;
  }
 
  /**
   * Calcula la confianza del anlisis emocional basado en coincidencia y pistas contextuales
   * @param {string} content - Contenido del mensaje
   * @param {Object} emotion - Emocin detectada
   * @returns {number} Confianza (0-1)
   */
  calculateConfidence(content, emotion) {
    if (!emotion || !this.isValidString(content)) {
      return this.CONFIDENCE_DEFAULT;
    }
    
    let confidence = this.CONFIDENCE_DEFAULT;
    
    // Alta confianza si hay coincidencia directa con patrones
    if (emotion.patterns?.test(content)) {
      confidence = this.CONFIDENCE_MATCH;
    }
    
    // Aumentar confianza si hay emojis que coinciden con la emocin detectada
    const emojiEmotion = this.detectEmotionFromEmojis(content);
    if (emojiEmotion === emotion.name) {
      confidence = Math.min(confidence + 0.15, this.CONFIDENCE_MAX);
    }
    
    // Aumentar confianza si hay pistas contextuales
    if (this.hasContextualClues(content)) {
      confidence = Math.min(confidence + this.CONFIDENCE_CONTEXTUAL, this.CONFIDENCE_MAX);
    }
    
    // Aumentar confianza si hay mltiples indicadores (intensificadores + emocin)
    if (this.intensifiersPattern.test(content) && emotion.patterns?.test(content)) {
      confidence = Math.min(confidence + 0.05, this.CONFIDENCE_MAX);
    }
    
    // Reducir confianza si hay negacin explcita de la emocin detectada
    if (this.denialPattern.test(content) && emotion.category === 'negative') {
      confidence = Math.max(confidence - 0.2, 0.3);
    }
    
    return Math.min(confidence, this.CONFIDENCE_MAX);
  }
 
  /**
   * Verifica si el contenido tiene pistas contextuales de emocin (expresiones de sentimiento)
   * @param {string} content - Contenido del mensaje
   * @returns {boolean} True si contiene pistas contextuales
   */
  hasContextualClues(content) {
    if (!this.isValidString(content)) {
      return false;
    }
    return this.contextualCluesPattern.test(content);
  }
 
  /**
   * Determina si la emocin requiere atencin especial (emociones negativas con alta intensidad)
   * @param {Object} emotion - Emocin detectada
   * @param {number} intensity - Intensidad emocional
   * @returns {boolean} True si requiere atencin especial
   */
  checkIfRequiresAttention(emotion, intensity) {
    if (!emotion) {
      return false;
    }
    return emotion.category === this.CATEGORY_NEGATIVE && 
           intensity >= this.INTENSITY_THRESHOLD_ATTENTION;
  }
 
  /**
   * Ajusta el anlisis emocional basado en el historial previo
   * @param {Object} currentEmotion - Emocin actual
   * @param {number} currentIntensity - Intensidad actual
   * @param {Array} previousPatterns - Historial previo
   * @returns {Object} Anlisis ajustado con emocin e intensidad
   */
  adjustBasedOnHistory(currentEmotion, currentIntensity, previousPatterns) {
    if (!Array.isArray(previousPatterns) || !previousPatterns.length) {
      return { emotion: currentEmotion, intensity: currentIntensity };
    }
    
    const recentPatterns = previousPatterns.slice(-this.HISTORY_WINDOW_SIZE);
    const emotionalTrend = this.analyzeEmotionalTrend(recentPatterns);
    
    return {
      emotion: currentEmotion,
      intensity: this.adjustIntensityBasedOnTrend(currentIntensity, emotionalTrend)
    };
  }
 
  /**
   * Analiza la tendencia emocional en el historial (aumentando, disminuyendo o estable)
   * @param {Array} patterns - Historial de anlisis
   * @returns {string} Tendencia ('increasing', 'decreasing', 'stable')
   */
  analyzeEmotionalTrend(patterns) {
    if (!Array.isArray(patterns) || !patterns.length) {
      return this.TREND_STABLE;
    }
    
    const intensities = patterns.map(p => p.intensity);
    const average = intensities.reduce((a, b) => a + b, 0) / intensities.length;
    const lastIntensity = intensities[intensities.length - 1];
    
    if (lastIntensity > average + this.INTENSITY_TREND_THRESHOLD) {
      return this.TREND_INCREASING;
    }
    if (lastIntensity < average - this.INTENSITY_TREND_THRESHOLD) {
      return this.TREND_DECREASING;
    }
    
    return this.TREND_STABLE;
  }
 
  /**
   * Ajusta la intensidad emocional segn la tendencia detectada en el historial
   * @param {number} intensity - Intensidad actual
   * @param {string} trend - Tendencia emocional ('increasing', 'decreasing', 'stable')
   * @returns {number} Intensidad ajustada (1-10)
   */
  adjustIntensityBasedOnTrend(intensity, trend) {
    switch (trend) {
      case this.TREND_INCREASING:
        return this.clampIntensity(intensity + this.INTENSITY_TREND_ADJUSTMENT);
      case this.TREND_DECREASING:
        return this.clampIntensity(intensity - this.INTENSITY_TREND_ADJUSTMENT);
      default:
        return intensity;
    }
  }
 
  /**
   * Devuelve un anlisis emocional por defecto cuando no se puede analizar el mensaje
   * @returns {Object} Anlisis por defecto con valores seguros
   */
  getDefaultAnalysis() {
    return {
      mainEmotion: this.EMOTION_NEUTRAL,
      intensity: this.INTENSITY_NEUTRAL,
      category: this.CATEGORY_NEUTRAL,
      secondary: [],
      confidence: this.CONFIDENCE_DEFAULT,
      requiresAttention: false,
      subtype: null,
      topic: 'general',
      topics: ['general']
    };
  }
