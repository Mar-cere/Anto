/**
 * Obtiene tcnicas inmediatas para una emocin especfica
 * @param {string} emotion - Emocin detectada
 * @param {number} intensity - Intensidad emocional (1-10)
 * @returns {Array} Array de tcnicas inmediatas
 */
export const getImmediateTechniques = (emotion, intensity = 5) => {
  const techniques = IMMEDIATE_TECHNIQUES[emotion] || [];
  
  // Si la intensidad es muy alta (8+), priorizar tcnicas de regulacin
  if (intensity >= 8) {
    return techniques.filter(t => 
      t.type === 'DBT' || 
      t.name.includes('Respiracin') || 
      t.name.includes('Grounding') ||
      t.name.includes('Tiempo Fuera')
    );
  }
  
  return techniques;
};

/**
 * Selecciona la tcnica ms apropiada segn emocin, intensidad y fase teraputica
 * @param {string} emotion - Emocin detectada
 * @param {number} intensity - Intensidad emocional (1-10)
 * @param {string} phase - Fase teraputica ('inicial', 'intermedia', 'avanzada')
 * @param {string} intent - Intencin del mensaje
 * @returns {Object|null} Tcnica seleccionada o null
 */
export const selectAppropriateTechnique = (emotion, intensity = 5, phase = 'inicial', intent = null) => {
  // Para intensidades muy altas (8+), priorizar tcnicas inmediatas de regulacin
  if (intensity >= 8) {
    const immediate = getImmediateTechniques(emotion, intensity);
    if (immediate.length > 0) {
      return {
        ...immediate[0],
        category: 'immediate',
        priority: 'high'
      };
    }
  }
 
  // Para fase inicial, usar tcnicas ms simples y accesibles
  if (phase === 'inicial') {
    const immediate = getImmediateTechniques(emotion, intensity);
    if (immediate.length > 0) {
      return {
        ...immediate[0],
        category: 'immediate',
        priority: 'medium'
      };
    }
  }
 
  // Para fases ms avanzadas, incluir tcnicas ms estructuradas
  if (phase === 'intermedia' || phase === 'avanzada') {
    // Combinar tcnicas inmediatas con tcnicas estructuradas
    const allTechniques = [
      ...getImmediateTechniques(emotion, intensity),
      ...getCBTTechniques(emotion),
      ...getDBTTechniques(emotion),
      ...getACTTechniques(emotion)
    ];
    
    if (allTechniques.length > 0) {
      // Seleccionar tcnica basada en la intencin si est disponible
      if (intent === 'SEEKING_HELP' || intent === 'CRISIS') {
        return {
          ...allTechniques[0],
          category: 'structured',
          priority: 'high'
        };
      }
      
      return {
        ...allTechniques[0],
        category: 'structured',
        priority: 'medium'
      };
    }
  }
 
  // Fallback: tcnica inmediata genrica
  const immediate = getImmediateTechniques(emotion, intensity);
  if (immediate.length > 0) {
    return {
      ...immediate[0],
      category: 'immediate',
      priority: 'low'
    };
  }
 
  return null;
};
 
/**
 * Formatea una tcnica como texto para incluir en respuestas del chat
 * @param {Object} technique - Objeto de tcnica
 * @param {Object} options - Opciones de formato { compact: boolean, maxSteps: number }
 * @returns {string} Texto formateado de la tcnica
 */
export const formatTechniqueForResponse = (technique, options = {}) => {
  if (!technique) return '';
 
  const { compact = false, maxSteps = 4 } = options;
 
  // Formato compacto para cuando hay limitaciones de espacio
  if (compact) {
    let text = `\n\n ${technique.name} (${technique.type || 'Teraputica'})\n\n`;
    
    if (technique.description) {
      text += `${technique.description}\n\n`;
    }
 
    // En formato compacto, mostrar solo los primeros 2-3 pasos
    if (technique.steps && Array.isArray(technique.steps) && technique.steps.length > 0) {
      const stepsToShow = Math.min(2, technique.steps.length);
      text += 'Pasos principales:\n';
      for (let i = 0; i < stepsToShow; i++) {
        text += `${i + 1}. ${technique.steps[i]}\n`;
      }
      if (technique.steps.length > stepsToShow) {
        text += `\nPuedes preguntarme por los pasos completos si te interesa.\n`;
      }
    }
 
    return text;
  }
 
  // Formato completo
  let text = `\n\n Tcnica sugerida: ${technique.name}\n`;
  text += `${technique.type || 'Teraputica'}\n\n`;
  
  if (technique.description) {
    text += `${technique.description}\n\n`;
  }
 
  if (technique.steps && Array.isArray(technique.steps) && technique.steps.length > 0) {
    text += 'Pasos:\n';
    const stepsToShow = Math.min(maxSteps, technique.steps.length);
    technique.steps.slice(0, stepsToShow).forEach((step, index) => {
      text += `${index + 1}. ${step}\n`;
    });
    
    // Si hay ms pasos, mencionarlo
    if (technique.steps.length > stepsToShow) {
      text += `\nHay ${technique.steps.length - stepsToShow} paso(s) adicional(es). Puedes preguntarme por los pasos completos.\n`;
    }
  }
 
  if (technique.whenToUse) {
    text += `\nCundo usar: ${technique.whenToUse}\n`;
  }
 
  return text;
};
